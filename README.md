            function resetPassword(thisObj) {
                var form = thisObj.closest('.form-password')
                var old_password = form.find('.js-old-password').val()
                var password = form.find('.js-new-password').val()
                var password_confirmation = form.find('.js-confirm-password').val()

                var button = thisObj
                button.prop('disabled', true)
                axios.post('/update-password', {
                    old_password: old_password,
                    password: password,
                    password_confirmation: password_confirmation,
                }).then(function (response) {
                    location.reload();
                }).catch(function (error) {
                    button.prop('disabled', false)
                    var text = '';
                    if (error.response.data.errors) {
                        var errors = error.response.data.errors
                        if (errors.old_password) {
                            text += errors.old_password[0] + '\n'
                        }
                        if (errors.password) {
                            text += errors.password[0] + '\n'
                        }
                        if (errors.password_confirmation) {
                            text += errors.password_confirmation[0] + '\n'
                        }
                    }
                    Swal.fire(
                        '',
                        text,
                        'error'
                    )
                });
            }
